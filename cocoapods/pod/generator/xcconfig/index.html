168 def add_developers_frameworks_if_needed(consumer, xcconfig)
169   if xcconfig.frameworks.include?('SenTestingKit')
170     search_paths = xcconfig.attributes['FRAMEWORK_SEARCH_PATHS'] ||= ''
171     DEVELOPER_FRAMEWORKS_SEARCH_PATHS.each do |search_path|
172       unless search_paths.include?(search_path)
173         search_paths << ' ' unless search_paths.empty?

176     end
