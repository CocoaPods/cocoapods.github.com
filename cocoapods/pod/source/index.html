79 def search(dependency)
80   pod_sets.find do |set|
81     # First match the (top level) name, which does not yet load the spec from disk
82     set.name == dependency.top_level_spec_name &&
83       # Now either check if it's a dependency on the top level spec, or if it's not

101 def search_by_name(query, full_text_search = false)
