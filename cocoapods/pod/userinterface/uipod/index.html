108 def distance_from_now_in_words(from_time)
109   return nil unless from_time
110   from_time = Time.parse(from_time)
111   to_time = Time.now
112   distance_in_days = (((to_time - from_time).abs)/60/60/24).round
113 
114   case distance_in_days
115   when 0..7
116     "less than a week ago"
117   when 8..29
118     "#{distance_in_days} days ago"
119   when 30..45
120     "1 month ago"
121   when 46..365
122     "#{(distance_in_days.to_f / 30).round} months ago"
123   else
124     "more than a year ago"
125   end
126 end
