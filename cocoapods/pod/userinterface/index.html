256 def wrap_string(txt, indent = '')
257   width = `stty size`.split(' ')[1].to_i - indent.length
258   width = 80 unless width >= 10
259   txt.strip.gsub(/(.{1,#{width}})( +|$)\n?|(.{#{width}})/, indent + "\\1\\3\n")
