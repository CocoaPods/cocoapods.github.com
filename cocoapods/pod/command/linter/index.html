53     end

211 def paths_starting_with_a_slash_errors
212   messages = []
213   %w[source_files public_header_files resources clean_paths].each do |accessor|
214     patterns = spec.send(accessor.to_sym)
215     # Some values are multiplaform
216     patterns = patterns.is_a?(Hash) ? patterns.values.flatten(1) : patterns
217     patterns = patterns.compact # some patterns may be nil (public_header_files, for instance)
218     patterns.each do |pattern|
219       # Skip FileList that would otherwise be resolved from the working directory resulting
220       # in a potentially very expensi operation
221       next if pattern.is_a?(FileList)
222       invalid = pattern.is_a?(Array) ? pattern.any? { |path| path.start_with?('/') } : pattern.start_with?('/')
223       if invalid
224         messages << "Paths cannot start with a slash (#{accessor})"
225         break
226       end
227     end
228   end
229   messages
230 end

279 def process_xcode_build_output(output)
280   output_by_line = output.split("\n")
281   selected_lines = output_by_line.select do |l|
282     l.include?('error: ') && (l !~ /errors? generated\./) && (l !~ /error: \(null\)/)\
283       || l.include?('warning: ') && (l !~ /warnings? generated\./)\
