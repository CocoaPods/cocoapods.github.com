217 def clean_paths
218   cached_used = used_files
219   glob_options = File::FNM_DOTMATCH | File::FNM_CASEFOLD
220   files = Pathname.glob(root + "**/*", glob_options).map(&:to_s)
221 
222   files.reject! do |candidate|
223     candidate = candidate.downcase
224     candidate.end_with?('.', '..') || cached_used.any? do |path|
225       path = path.downcase
226       path.include?(candidate) || candidate.include?(path)
227     end
228   end
229   files
230 end
